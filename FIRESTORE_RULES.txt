// FIRESTORE SECURITY RULES
// Copy-paste this entire ruleset into Firebase Console > Firestore > Rules
// Then click "Publish"

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ==================== USERS COLLECTION ====================
    // Users Collection - Approval workflow
    match /users/{userId} {
      // Users can read their own document
      allow read: if request.auth.uid == userId;
      
      // Users can create their own pending document during signup.
      // Admins can also create approved user profiles directly from the admin console.
      allow create: if (
        request.auth.uid == userId 
        && request.resource.data.uid == userId
        && request.resource.data.status == 'pending'
        && !request.resource.data.keys().hasAny(['approvedAt', 'approvedBy'])
        && (
          !request.resource.data.keys().hasAny(['role'])
          || request.resource.data.role == 'agent'
        )
      ) || (
        isAdmin()
        && request.resource.data.uid == userId
        && request.resource.data.status == 'approved'
        && request.resource.data.approvedBy == request.auth.uid
      );
      
      // Users can update only their own document (except status/role which only admin updates)
      allow update: if request.auth.uid == userId 
        && !request.resource.data.diff(resource.data).affectedKeys().hasAny(['status', 'role', 'approvedAt', 'approvedBy']);
      
      // Admins can read all users
      allow read: if isAdmin();
      
      // Admins can update status, role, approval fields
      allow update: if isAdmin();
      
      // Admins can delete user documents when removing agents/users
      allow delete: if isAdmin();
      
      match /notificationLogs/{logId} {
        allow create: if request.auth.uid == userId;
        allow read: if request.auth.uid == userId || isAdmin();
      }
      
      match /data/{dataId} {
        allow read, create, update, delete: if request.auth.uid == userId || isAdmin();
      }
    }
    
    match /roles/{roleId} {
      // Only admin users can manage role definitions
      allow read: if isAdmin();
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // ==================== AGENTS COLLECTION ====================
    // Agents Collection - User agents management
    match /agents/{agentId} {
      // Users can read agents they own; admins can manage all agents.
      allow read: if isAdmin() || request.auth.uid == resource.data.ownerId;
      
      // Approved users can create agents
      allow create: if request.auth != null 
        && isApproved()
        && request.resource.data.ownerId == request.auth.uid;
      
      // Users can update agents they own
      allow update: if isAdmin() || request.auth.uid == resource.data.ownerId;
      
      // Users can delete agents they own
      allow delete: if isAdmin() || request.auth.uid == resource.data.ownerId;
      
      match /data/{dataId} {
        allow read, create, update, delete: if request.auth.uid == agentId || isAdmin();
      }
    }
    
    // ==================== UMRAH QUOTATIONS ====================
    match /umrah_quotations/{ownerId} {
      allow read: if request.auth != null && (ownerId == request.auth.uid || isAdmin());

      allow create: if (isApproved() && ownerId == request.auth.uid || isAdmin())
        && isValidQuotationDocument(ownerId);

      allow update: if (isAdmin() || (isApproved() && ownerId == request.auth.uid))
        && isValidQuotationDocument(ownerId);

      allow delete: if request.auth != null && (ownerId == request.auth.uid || isAdmin());
    }
    
    // ==================== INTERNATIONAL QUOTATIONS ====================
    match /international_quotations/{ownerId} {
      allow read: if request.auth != null && (ownerId == request.auth.uid || isAdmin());

      allow create: if (isApproved() && ownerId == request.auth.uid || isAdmin())
        && isValidQuotationDocument(ownerId);

      allow update: if (isAdmin() || (isApproved() && ownerId == request.auth.uid))
        && isValidQuotationDocument(ownerId);

      allow delete: if request.auth != null && (ownerId == request.auth.uid || isAdmin());
    }

    // ==================== DOMESTIC QUOTATIONS ====================
    match /domestic_quotations/{ownerId} {
      allow read: if request.auth != null && (ownerId == request.auth.uid || isAdmin());

      allow create: if (isApproved() && ownerId == request.auth.uid || isAdmin())
        && isValidQuotationDocument(ownerId);

      allow update: if (isAdmin() || (isApproved() && ownerId == request.auth.uid))
        && isValidQuotationDocument(ownerId);

      allow delete: if request.auth != null && (ownerId == request.auth.uid || isAdmin());
    }

    // ==================== QUOTATION LINKS COLLECTION ====================
    // Public quotation share links
    match /quotation_links/{linkId} {
      allow get: if true;
      allow create: if request.auth != null && (request.resource.data.ownerId == request.auth.uid || isAdmin());
      allow update: if request.auth != null && (
        isAdmin()
        || (
          resource.data.ownerId == request.auth.uid
          && request.resource.data.ownerId == resource.data.ownerId
        )
      );
      allow delete: if request.auth != null && (resource.data.ownerId == request.auth.uid || isAdmin());
    }

    // ==================== CUSTOMERS COLLECTION ====================
    // Customers - shared CRM records generated from quotations
    match /customers/{customerId} {
      // Approved users can read customers for lookup in quotation tools
      allow read: if isApproved();

      // Approved users can create and update customer profiles
      allow create: if isApproved();
      allow update: if isApproved();

      // Only admins can delete customer profiles
      allow delete: if isAdmin();
    }
    
    // ==================== PACKAGES COLLECTION ====================
    // Packages - Authenticated users can read, admin can write
    match /packages/{packageId} {
      // Authenticated users can read all packages
      allow read: if request.auth != null;
      
      // Admin users can create packages
      allow create: if isAdmin();
      
      // Admin users can update packages
      allow update: if isAdmin();
      
      // Admin users can delete packages
      allow delete: if isAdmin();
    }
    
    // ==================== DESTINATIONS COLLECTION ====================
    // Destinations - For package builder reference data
    match /destinations/{destinationId} {
      // Anyone can read destinations
      allow read: if true;
      
      // Only admin can modify destinations
      allow write: if isAdmin();
    }

    // ==================== VOUCHERS COLLECTION ====================
    // Vouchers - agents create/manage, customers view published/cancelled vouchers
    match /vouchers/{voucherId} {
      // Owner/admin can read voucher docs.
      allow read: if isAdmin() || (isApproved() && isVoucherOwner(resource.data));

      // Public customer access to published/cancelled vouchers by direct ID.
      // (No list rule here; only direct document get is expected by app.)
      allow get: if resource.data.status in ['published', 'cancelled'];

      // Owners can create/edit their own vouchers, admins can manage all.
      allow create: if isAdmin() || (isApproved() && isVoucherOwner(request.resource.data));
      allow update: if isAdmin() || (
        isApproved()
        && isVoucherOwner(resource.data)
        && isVoucherOwner(request.resource.data)
      );

      // Public view logging from voucher-view page (increments viewCount + viewedAt only).
      allow update: if resource.data.status in ['published', 'cancelled']
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['viewCount', 'viewedAt'])
        && request.resource.data.viewCount >= resource.data.viewCount;

      // Restrict hard delete to admin.
      allow delete: if isAdmin();
    }
    
    // ==================== INTERNATIONAL VOUCHERS COLLECTION ====================
    // Int Vouchers - agents create/manage, customers view published/cancelled vouchers
    match /int_vouchers/{voucherId} {
      // Owner/admin can read voucher docs.
      allow read: if isAdmin() || (isApproved() && isVoucherOwner(resource.data));

      // Public customer access to published/cancelled vouchers by direct ID.
      allow get: if resource.data.status in ['published', 'cancelled'];

      // Owners can create/edit their own vouchers, admins can manage all.
      allow create: if isAdmin() || (isApproved() && isVoucherOwner(request.resource.data));
      allow update: if isAdmin() || (
        isApproved()
        && isVoucherOwner(resource.data)
        && isVoucherOwner(request.resource.data)
      );

      // Public view logging from voucher-view page (increments viewCount + viewedAt only).
      allow update: if resource.data.status in ['published', 'cancelled']
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['viewCount', 'viewedAt'])
        && request.resource.data.viewCount >= resource.data.viewCount;

      // Restrict hard delete to admin.
      allow delete: if isAdmin();
    }

    // ==================== DOMESTIC VOUCHERS COLLECTION ====================
    // Domestic Vouchers - agents create/manage, customers view published/cancelled vouchers
    match /domestic_vouchers/{voucherId} {
      // Owner/admin can read voucher docs.
      allow read: if isAdmin() || (isApproved() && isVoucherOwner(resource.data));

      // Public customer access to published/cancelled vouchers by direct ID.
      allow get: if resource.data.status in ['published', 'cancelled'];

      // Owners can create/edit their own vouchers, admins can manage all.
      allow create: if isAdmin() || (isApproved() && isVoucherOwner(request.resource.data));
      allow update: if isAdmin() || (
        isApproved()
        && isVoucherOwner(resource.data)
        && isVoucherOwner(request.resource.data)
      );

      // Public view logging from voucher-view page (increments viewCount + viewedAt only).
      allow update: if resource.data.status in ['published', 'cancelled']
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['viewCount', 'viewedAt'])
        && request.resource.data.viewCount >= resource.data.viewCount;

      // Restrict hard delete to admin.
      allow delete: if isAdmin();
    }
    
    // ==================== HELPER FUNCTIONS ====================
    
    // Check if user is admin
    function isAdmin() {
      return request.auth != null 
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // Check if user is approved
    function isApproved() {
      return request.auth != null 
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.status == 'approved';
    }
    
    // Check if user document exists
    function userExists(userId) {
      return exists(/databases/$(database)/documents/users/$(userId));
    }

    function isValidQuotationDocument(ownerId) {
      return request.resource.data.agentId == ownerId
        && request.resource.data.userId == ownerId
        && request.resource.data.quotations is list;
    }

    // Check if voucher belongs to the authenticated user
    function isVoucherOwner(voucherData) {
      return request.auth != null
        && voucherData.agentData != null
        && (
          voucherData.agentData.id == request.auth.uid
          || voucherData.agentData.uid == request.auth.uid
          || voucherData.agentData.userId == request.auth.uid
        );
    }
    
  }
}
